def decrypt_rot_16(text):
    return ''.join([chr((ord(char) - 65 - 16) % 26 + 65) if char.isalpha() and char.isupper() else
                    chr((ord(char) - 97 - 16) % 26 + 97) if char.isalpha() and char.islower() else char
                    for char in text])
